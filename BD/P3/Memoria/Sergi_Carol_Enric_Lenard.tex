\begin{lstlisting}

FROM empleat 
WHERE id_empleat IN

	FROM feina 
	WHERE id_empresa != 12

		FROM feina

		); 
\end{lstlisting}

\subsection{Troba la empresa que t ́e m ́es empleats}

\begin{lstlisting}

FROM feina 
GROUP BY id_empresa 
ORDER BY count DESC
LIMIT 1;
\end{lstlisting}
